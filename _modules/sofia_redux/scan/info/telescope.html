[docs]class TelescopeInfo(InfoBase):

    def __init__(self):
        """
        Initialize the telescope information.

        Contains information on the telescope, telescope coordinate system
        epoch, and whether it is tracking the source coordinates.
        """
        super().__init__()
        self.is_tracking = False
        self.epoch = J2000
        self.telescope = None

    @property
    def log_id(self):
        """
        Return the string log ID for the info.

        The log ID is used to extract certain information from table data.

        Returns
        -------
        str
        """
        return 'tel'
